In your Javascript

$('#fb-connect').click(function() {
    OAuth.initialize('YOUR_PUBLIC_KEY');
    OAuth.popup('facebook', function(err, res) {
        if (err) {
            // do something with error
            alert(JSON.stringify(err));
            return;
        }
        $('#token').html(res.access_token)
                   .parent().show();
    });
});
